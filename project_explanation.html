def extract_landmarks(image_path, landmarker): img = cv2.imread(image_path) if img is None: return None rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB) mp_image = mp.Image(image_format=mp.ImageFormat.SRGB, data=rgb) result = landmarker.detect(mp_image) if not result.hand_landmarks: return None landmarks = result.hand_landmarks[0] wrist = landmarks[0] coords = [] for lm in landmarks: coords.extend([lm.x - wrist.x, lm.y - wrist.y, lm.z - wrist.z]) return coords

  • result.hand_landmarks[0] — רשימה של 21 נקודות של היד הראשונה שזוהתה.

מה זה נרמול (Normalization)?

אם נשמור את הקואורדינטות המוחלטות (x=0.3, y=0.5), המודל ייתן ניחושים שונים לאותה אות תלוי אם היד נמצאת בצד שמאל או ימין של המסך — לא טוב.

lm.x - wrist.x — מחסירים את מיקום פרק כף היד מכל נקודה. אחרי זה כל הקואורדינטות הן יחסיות לפרק כף היד. עכשיו לא חשוב איפה היד על המסך — הערכים יהיו זהים.

תוצאה: 21 נקודות × 3 ערכים (x, y, z) = 63 מספרים לכל תמונה.

  • coords.extend([...]) — extend מוסיפה מספר ערכים לרשימה בבת אחת (בניגוד ל-append שמוסיף ערך אחד).

main() — בניית הדאטאסט (שורות 62–88)

X = [] y = [] skipped = 0 for label_idx, (heb, name) in enumerate(LETTERS): folder = os.path.join(DATA_DIR, name) images = [f for f in os.listdir(folder) if f.lower().endswith('.jpg')] for img_file in images: features = extract_landmarks(os.path.join(folder, img_file), landmarker) if features: X.append(features) y.append(label_idx) else: skipped += 1

בניית הדאטאסט לאימון:

  • X — רשימת הfeatures. כל שורה = 63 מספרים (תמונה אחת). X הוא קונבנציה ב-ML לנתוני הקלט.
  • y — רשימת התוויות. כל ערך = מספר האות (0=א, 1=ב...). y הוא קונבנציה ב-ML לתשובה הנכונה.
  • enumerate(LETTERS) — עוברת על הרשימה ומספקת גם אינדקס (0,1,2...) וגם הערך. label_idx יהיה 0 עבור א, 1 עבור ב וכו'.
  • אחרי הלולאה: X הוא מטריצה של [4394 × 63], y הוא וקטור של [4394].
X = np.array(X) y = np.array(y) min_samples = 5 valid_mask = np.array([np.sum(y == i) >= min_samples for i in y]) X, y = X[valid_mask], y[valid_mask]
  • np.array(X) — ממיר רשימת פייתון למערך NumPy. חובה לפני שימוש ב-sklearn.
  • valid_mask — מסנן אותיות עם פחות מ-5 דגימות. np.sum(y == i) = כמה פעמים i מופיע ב-y. אם פחות מ-5 — לא ניתן לחלק לtrain/test.
  • X[valid_mask] — ב-NumPy אפשר לסנן מערך עם מערך של True/False. נקרא Boolean Indexing.

  • train_test_split(..., test_size=0.2) — מחלק 80% לאימון, 20% לבדיקה. random_state=42 מבטיח שהחלוקה תהיה זהה בכל הרצה (reproducibility).
  • RandomForestClassifier(n_estimators=100) — יוצר 100 עצי החלטה. כל עץ מאומן על subset שונה של הדאטא ומצביע על ניחוש. התוצאה הסופית = הצבעת רוב.
  • n_jobs=-1 — משתמש בכל ליבות המעבד לאימון מקבילי (מהיר יותר).
  • model.fit(X_train, y_train) — זה האימון עצמו. המודל לומד את הקשר בין 63 המספרים לאות.
  • model.predict(X_test) — מנחש את האות לכל דגימה בסט הבדיקה.
  • pickle.dump({'model': model, 'letters': LETTERS}, f) — שומר שני דברים לקובץ: המודל עצמו, ורשימת האותיות (כדי שrecognize.py ידע איזה מספר = איזה אות).

מה זה Random Forest?

דמייני שאת שואלת 100 אנשים שונים "איזו אות זאת?" — כל אחד מסתכל על היד קצת אחרת. התשובה הסופית היא מה שרוב האנשים אמרו. כך עובד Random Forest — 100 "עצי החלטה" שכל אחד לומד קצת אחרת, והניחוש הסופי הוא הצבעת הרוב.

📁 recognize.py — זיהוי בזמן אמת

פונקציה: get_features() (שורות 46–51)

def get_features(landmarks): wrist = landmarks[0] coords = [] for lm in landmarks: coords.extend([lm.x - wrist.x, lm.y - wrist.y, lm.z - wrist.z]) return coords

זהה לחלוטין ל-extract_landmarks מ-train_model.py, אבל פשוטה יותר כי קלט הוא landmarks שכבר זוהו (לא תמונה). מחשבת את אותם 63 מספרים מנורמלים שהמודל ציפה לקבל באימון.

חשוב: חייבים להכין features בדיוק באותה דרך כמו באימון. אחרת המודל יקבל קלט שלא מוכר לו.

מה זה Smoothing ולמה צריך?

המצלמה מצלמת ~30 פריימים בשנייה. בכל פריים המודל מנחש. בגלל רעש קטן בתמונה, הניחוש עשוי לקפוץ: א, א, ב, א, א — הב הזה הוא רעש.

הפתרון: שומרים 7 ניחושים אחרונים ומציגים מה שמופיע הכי הרבה. אם 6 מתוך 7 אומרים א — מציגים א. הניחוש יציב ולא מרצד.

לולאת הזיהוי הראשית (שורות 82–146)

if hand_ok: draw_hand(frame, result.hand_landmarks[0]) features = get_features(result.hand_landmarks[0]) proba = model.predict_proba([features])[0] label_idx = int(np.argmax(proba)) confidence = proba[label_idx] history.append(label_idx) if len(history) > HISTORY_SIZE: history.pop(0) stable_idx = max(set(history), key=history.count) predicted_heb, predicted_name = letters[stable_idx]
  • model.predict_proba([features]) — במקום להחזיר רק את האות הכי סבירה, מחזיר הסתברות לכל 22 האותיות. למשל: [0.02, 0.01, 0.95, ...] — 95% אות ג, 2% אות א וכו'.
  • [features] — עוטפים ב-[ ] כי predict_proba מצפה לרשימה של דגימות (אפילו אם יש רק אחת).
  • [0] — לוקחים את הדגימה הראשונה (ויחידה).
  • np.argmax(proba) — מחזיר את האינדקס של הערך הגבוה ביותר. זה אינדקס האות עם ההסתברות הגבוהה ביותר.
  • confidence = proba[label_idx] — ה"ביטחון" — ההסתברות של האות שנבחרה (בין 0 ל-1).
  • history.pop(0) — מסיר את הניחוש הישן ביותר. pop(0) = מסיר מהתחלה.
  • max(set(history), key=history.count) — מוצא את הערך שמופיע הכי הרבה פעמים ב-history. set(history) = כל הערכים הייחודיים. key=history.count = ממיין לפי כמה פעמים כל ערך מופיע.
bar_w = int((w - 200) * confidence) cv2.rectangle(frame, (160, 20), (w - 20, 50), (50, 50, 50), -1) bar_color = (0, 220, 80) if confidence > 0.7 else (0, 140, 255) cv2.rectangle(frame, (160, 20), (160 + bar_w, 50), bar_color, -1)
  • bar_w = (w - 200) * confidence — רוחב הbbar לפי הביטחון. ביטחון=1.0 → bar מלא. ביטחון=0.5 → bar חצי.
  • (0, 220, 80) if confidence > 0.7 else (0, 140, 255) — ternary operator בפייתון. ביטחון מעל 70% = ירוק, מתחת = כחול-כתום. זה נותן פידבק ויזואלי מיידי.

  • כשמייבאים קובץ מקובץ אחר (import collect_data) — __name__ = 'collect_data' → main() לא רץ אוטומטית.
